207 // Geh von der aktuellen Position zu Position x/y
208 // t ist keyw_walk oder keyw_jump ==> mit / ohne Strich zeichnen
209 static void walk(double x, double y, type_t t)
210 {
211  //printf("%f/%f - %f/%f %d\n", g_x, g_y, x, y, t == keyw_walk);
212  if (t == keyw_walk) {
213  int x1, y1, x2, y2;
214  if (to_x_pixel(g_x, &x1) && to_y_pixel(g_y, &y1) &&
215  to_x_pixel(x, &x2) && to_y_pixel(y, &y2)) {
216  sdlDrawLine(x1, y1, x2, y2,
217  to_color(g_red), to_color(g_green), to_color(g_blue));
218  sdlUpdate();
219  sdlMilliSleep(g_delay);
220  } else {
221  printf("Line %.2f/%.2f - %.2f/%.2f is offscreen\n", g_x, g_y, x, y);
222  }
223  }
224  g_x = x; g_y = y;
225  g_dist = sqrt(x * x + y * y);
226 }

247 // Liefert einen Pointer auf den Wert der Variablen t
248 // Bei lokalen Variablen:
249 // Eine Instanz von t im aktuellen Stack-Frame muss bereits existieren
250 // ro: Wird die Variable im Aufrufer nur gelesen oder auch geschrieben?
251 static double *var_ptr(const treenode_t *t, bool ro)
252 {
253  nameentry_t *n = t->d.p_name;
254  switch (n->type) {
255  case name_var:
256  // Stack vom Top abwärts durchsuchen,
257  // höchstens bis zum NULL-Eintrag,
258  // der die Variablen und Parameter der aktuellen Funktion begrenzt
259  for (vstack_elem_t *p = stack_top; p->name != NULL; --p) {
260  if (p->name == n) {
261  return &(p->val);
262  }
263  }
264  code_error(&(t->pos), "Variable %s hat noch keinen Wert\n", n->name);
265  break;
266  case name_glob:
267  return &(n->d.val);
268  break;
269  case name_pvar_ro:
270  assert(ro);
271  return n->d.p_val;
272  break;
273  case name_pvar_rw:
274  return n->d.p_val;
275  break;
276  default:
277  assert(false);
278  }
279 }

281 // Rechnet exp aus und speichert das Ergebnis in der Variable t
282 // Liefert einen Pointer auf ihren Wert
283 // Bei lokalen Variablen:
284 // Gibt es die Variable schon im lokalen Stackframe,
285 // wird die bestehende Instanz verwendet, sonst eine neue angelegt
286 static double *set_var(const treenode_t *t, const treenode_t *exp)
287 {
288  nameentry_t *n = t->d.p_name;
289  double x = expr(exp);
290  switch (n->type) {
291  case name_var:
292  // Stack vom Top abwärts durchsuchen,
293  // höchstens bis zum NULL-Eintrag,
294  // der die Variablen und Parameter der aktuellen Funktion begrenzt
295  for (vstack_elem_t *p = stack_top; p->name != NULL; --p) {
296  if (p->name == n) {
297  p->val = x;
298  return &(p->val);
299  }
300  }
301  // Die Variable gibt es im Stackframe noch nicht: Anlegen
302  push(n, x, &(t->pos));
303  return &(stack_top->val);
304  break;
305  case name_glob:
306  n->d.val = x;
307  return &(n->d.val);
308  break;
309  case name_pvar_ro:
310  assert(false); // readonly!
311  break;
312  case name_pvar_rw:
313  *(n->d.p_val) = x;
314  return n->d.p_val;
315  break;
316  default:
317  assert(false);
318  }
319 }

334 // Legt einen Eintrag auf den Stack
335 // Macht den Stack bei Bedarf größer
336 // Mit var gleich NULL auch für den Funktions-Trenn-Eintrag verwendet
337 static void push(const nameentry_t *var, double val, const srcpos_t *pos)
338 {
339  if (var != NULL) assert(var->type == name_var);
340  ++stack_top;
341  if (stack_top == stack_end) {
342  // Stack ist voll, Größe verdoppeln
343  ptrdiff_t fill = stack_top - stack;
344  ptrdiff_t size = 2 * fill;
345  stack = realloc(stack, size * sizeof(vstack_elem_t));
346  mem_check(stack, "weitere Variablen", pos);
347  stack_top = stack + fill;
348  stack_end = stack + size;
349  }
350  stack_top->name = var; stack_top->val = val;
351 }

363 // Prüft, ob die Argument-Anzahl im Knoten t gleich expected ist
364 // Wenn nicht: Fehlermeldung ausgeben und Programm beenden
365 static void check_arg_cnt(const treenode_t *t, int expected)
366 {
367  for (int actual = 0; ; ++actual) {
368  if ((actual == MAX_ARGS) || (t->son[actual] == NULL)) {
369  if (actual != expected) {
370  code_error(&(t->pos), "Die Anzahl der Werte im Auruf (%d) "
371  "passt nicht zur Anzahl der Parameter (%d)",
372  actual, expected);
373  }
374  return;
375  }
376  }
377 }

379 // Beginnt einen Funktions- oder Pfadaufruf:
380 // - Variablen-Stackframe anlegen
381 // - Parameter ausrechnen und anlegen
382 static void start_call(const funcdef_t *f, const treenode_t *t)
383 {
384  int cnt = param_cnt(f);
385  check_arg_cnt(t, cnt);
386 
387  // zuerst alle Argumente im *alten* Stack-Frame ausrechnen
388  double args[MAX_ARGS];
389  for (int i = 0; i < cnt; ++i) {
390  args[i] = expr(t->son[i]);
391  }
392  // Ein Stack-Eintrag mit Namenspointer NULL beginnt ein neues Stack-Frame
393  push(NULL, 0, &(t->pos));
394  // dann die Parameter auf dem Stack anlegen
395  for (int i = 0; i < cnt; ++i) {
396  push(f->params[i], args[i], &(t->pos));
397  }
398 }

400 // Räumt den Var-Stack am Ende eines Funktions- oder Pfadaufrufs wieder auf:
401 // Entfernt die lokalen Variablen und Parameter vom Stack
402 static void end_call(void)
403 {
404  // Entferne alle Einträge des aktuellen Stack-Frames bis zum NULL-Element
405  while (stack_top->name != NULL) {
406  --stack_top; // vom Stack entfernen
407  }
408  // Ein Eintrag ohne Namen markiert das Ende des aktuellen Stack-Frames
409  assert(stack_top > stack); // das unterste Element muss bleiben!
410  --stack_top; // Markierungselement auch entfernen
411 }

413 // Führt einen Calc-Funktions-Aufruf aus
414 static double fcall(const treenode_t *t)
415 {
416  const nameentry_t *n = t->d.p_name;
417  switch (n->type) {
418  case name_calc: {
419  const funcdef_t *fdef = n->d.func;
420  start_call(fdef, t);
421  if (fdef->body != NULL) slist(fdef->body);
422  double result = expr(fdef->ret);
423  end_call();
424  return result;
425  break;
426  }
427  case name_math_sin:
428  case name_math_cos:
429  case name_math_tan:
430  check_arg_cnt(t, 1);
431  return (n->d.math)(expr(t->son[0]) * (2 * M_PI / 360));
432  break;
433  case name_math_sqrt:
434  check_arg_cnt(t, 1);
435  return (n->d.math)(expr(t->son[0]));
436  break;
437  case name_math_rand: {
438  check_arg_cnt(t, 2);
439  double lower = expr(t->son[0]);
440  double upper = expr(t->son[1]);
441  double r = rand() / ((double)RAND_MAX);
442  return lower + r * (upper - lower);
443  break;
444  }
445  default:
446  assert(false);
447  }
448 }

499 // Rechne eine Rechnung aus
500 static double expr(const treenode_t *t)
501 {
502  assert(t);
503  switch (t->type) {
504  case name_any:
505  return *(var_ptr(t, true));
506  case oper_const:
507  return t->d.val;
508  case oper_neg:
509  return -expr(t->son[0]);
510  case oper_pow:
511  return pow(expr(t->son[0]), expr(t->son[1]));
512  case oper_mul:
513  return expr(t->son[0]) * expr(t->son[1]);
514  case oper_div:
515  return expr(t->son[0]) / expr(t->son[1]);
516  case oper_add:
517  return expr(t->son[0]) + expr(t->son[1]);
518  case oper_sub:
519  return expr(t->son[0]) - expr(t->son[1]);
520  case oper_abs:
521  return fabs(expr(t->son[0]));
522  case oper_lpar:
523  return fcall(t);
524  default:
525  assert(false);
526  }
527 }

529 // Rechne eine Bedingung aus
530 static bool cond(const treenode_t *t)
531 {
532  assert(t);
533  switch (t->type) {
534  case oper_equ:
535  return eps_equal(expr(t->son[0]), expr(t->son[1]));
536  case oper_nequ:
537  return !eps_equal(expr(t->son[0]), expr(t->son[1]));
538  case oper_less:
539  return expr(t->son[0]) < expr(t->son[1]);
540  case oper_lequ:
541  return expr(t->son[0]) <= expr(t->son[1]);
542  case oper_grtr:
543  return expr(t->son[0]) > expr(t->son[1]);
544  case oper_gequ:
545  return expr(t->son[0]) >= expr(t->son[1]);
546  case keyw_not:
547  return !cond(t->son[0]);
548  case keyw_and:
549  return cond(t->son[0]) && cond(t->son[1]);
550  case keyw_or:
551  return cond(t->son[0]) || cond(t->son[1]);
552  default:
553  assert(false);
554  }
555 }

557 // Arbeite eine Liste von Statements ab
558 static void slist(const treenode_t *t)
559 {
560  assert(t);
561  for ( ; t; t = t->next) {
562  switch (t->type) {
563  case keyw_walk:
564  case keyw_jump:
565  if (t->d.walk == keyw_home) {
566  walk(0, 0, t->type);
567  g_dir = 0;
568  } else if (t->d.walk == keyw_mark) {
569  walk(top_mark->x, top_mark->y, t->type);
570  g_dir = top_mark->dir;
571  pop_mark();
572  } else {
573  double dist = expr(t->son[0]);
574  if (t->d.walk == keyw_back) {
575  dist = -dist;
576  } else {
577  assert(t->d.walk == keyw_walk);
578  }
579  walk(g_x + dist * cos(to_rad(g_dir)),
580  g_y + dist * sin(to_rad(g_dir)),
581  t->type);
582  }
583  break;
584  case keyw_left:
585  g_dir += expr(t->son[0]);
586  g_dir = to_range(g_dir);
587  break;
588  case keyw_right:
589  g_dir -= expr(t->son[0]);
590  g_dir = to_range(g_dir);
591  break;
592  case keyw_direction:
