177// Im Syntaxbaum werden folgende Knotentypen verwendet,
178// mit folgenden Unter-Syntaxbäumen im son-Array (keine wenn nichts angegeben)
179// und folgender Information im Art-abhängigen Member d
180//
181// name_any Variable (lesend in einer Rechnung): d = Namenseintrag (p_name)
182// oper_const Zahl: Konstante, d = Wert (val)
183// oper_neg unäres -: son 0 = Operand
184// oper_pow, oper_mul, oper_div, oper_add, oper_sub
185// ^ * / + -: son 0 und son 1 = Operanden
186// oper_abs, Absolutbetrag: son 0 = Operand
187// oper_lpar '(' = Funktionsaufruf:
188// d = Namenseintrag der Funktion (p_name)
189// son 0, 1, ...: Argumente (nicht vorhandene Argumente sind NULL)
190// oper_equ, oper_nequ, oper_less, oper_lequ, oper_grtr, oper_gequ
191// = <> < <= > >=: son 0 und son 1 = Operanden
192// oper_not Logisches not: son 0 = Operand
193// oper_and, oper_or Logisches and und or: son 0 und son 1 = Operanden
194// keyw_walk, keyw_jump
195// walk oder jump in allen Kombinationen
196// d = Art des walk: keyw_walk, keyw_back, keyw_home, keyw_mark
197// bei walk/jump und walk/jump back: son 0 = expr
198// keyw_right turn und turn right: son 0 = expr
199// keyw_left turn left: son 0 = expr
200// keyw_direction direction: son 0 = expr
201// keyw_clear, keyw_stop, keyw_finish, keyw_mark
202// clear, stop, finish, mark: Keine Operanden
203// keyw_path path-Aufruf: d = Name der Pfadfunktion, son 0, 1, ...: Argumente
204// nicht vorhandene Argumente sind NULL
205// keyw_color color: son 0 = rot, son 1 = grün, son 2 = blau
206// keyw_store, keyw_add, keyw_sub, keyw_mul, keyw_div
207// store, add, sub, mul, div:
208// d = Namens-Eintrag der Variable (p_name),
209// son 0 = expr
210// keyw_if if: son 0 = cond, son 1 = then-statements,
211// son 2 = else-statements oder NULL
212// keyw_do, times-Schleife: son 0 = expr, son 1 = statements
213// keyw_counter counter-Schleife:
214// d = Namens-Eintrag der Variable (p_name),
215// son 0 = Startwert
216// to: son 1 = Endwert, son 2 = NULL
217// downto: son 1 = NULL, son 2 = Endwert
218// son 3 = Schrittwert oder NULL
219// son 4 = statements
220// keyw_while while-Schleife: son 0 = cond, son 1 = statements
221// keyw_repeat repeat-Schleife: son 0 = cond, son 1 = statements
222//
223// Vorrang-Klammern in Rechnungen und Bedingungen
224// sind *keine* eigenen Knoten im Syntaxbaum
225// sondern beeinflussen beim Parsen nur die Struktur des Syntaxbaumes
226
227// Typ eines Syntaxbaum-Knotens
228struct _treenode {
229 type_t type; // Art des Knotens
230 srcpos_t pos; // Position im Sourcefile (für Fehlermeldungen)
231 treenode_t *next; // Bei allen Statements: Nächstes Statement
232 // (oder NULL beim letzten Statement einer Statement-Liste)
233 // Bei allen anderen Knoten: NULL
234 treenode_t *son[MAX_ARGS]; // Sohn-Syntaxbäume
235 nodedata_t d; // Daten zum Knoten
236};
