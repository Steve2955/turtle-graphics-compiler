193 // Im Syntaxbaum werden folgende Knotentypen verwendet,
194 // mit folgenden Unter-Syntaxbäumen im son-Array (keine wenn nichts angegeben)
195 // und folgender Information im Art-abhängigen Member d
196 //
197 // name_any Variable (lesend in einer Rechnung): d = Namenseintrag (p_name)
198 // oper_const Zahl: Konstante, d = Wert (val)
199 // oper_neg unäres -: son 0 = Operand
200 // oper_pow, oper_mul, oper_div, oper_add, oper_sub
201 // ^ * / + -: son 0 und son 1 = Operanden
202 // oper_abs, Absolutbetrag: son 0 = Operand
203 // oper_lpar '(' = Funktionsaufruf:
204 // d = Namenseintrag der Funktion (p_name)
205 // son 0, 1, ...: Argumente (nicht vorhandene Argumente sind NULL)
206 // oper_equ, oper_nequ, oper_less, oper_lequ, oper_grtr, oper_gequ
207 // = <> < <= > >=: son 0 und son 1 = Operanden
208 // oper_not Logisches not: son 0 = Operand
209 // oper_and, oper_or Logisches and und or: son 0 und son 1 = Operanden
210 // keyw_walk, keyw_jump
211 // walk oder jump in allen Kombinationen
212 // d = Art des walk: keyw_walk, keyw_back, keyw_home, keyw_mark
213 // bei walk/jump und walk/jump back: son 0 = expr
214 // keyw_right turn und turn right: son 0 = expr
215 // keyw_left turn left: son 0 = expr
216 // keyw_direction direction: son 0 = expr
217 // keyw_clear, keyw_stop, keyw_finish, keyw_mark
218 // clear, stop, finish, mark: Keine Operanden
219 // keyw_path path-Aufruf: d = Name der Pfadfunktion, son 0, 1, ...: Argumente
220 // nicht vorhandene Argumente sind NULL
221 // keyw_color color: son 0 = rot, son 1 = grün, son 2 = blau
222 // keyw_store, keyw_add, keyw_sub, keyw_mul, keyw_div
223 // store, add, sub, mul, div:
224 // d = Namens-Eintrag der Variable (p_name),
225 // son 0 = expr
226 // keyw_if if: son 0 = cond, son 1 = then-statements,
227 // son 2 = else-statements oder NULL
228 // keyw_do, times-Schleife: son 0 = expr, son 1 = statements
229 // keyw_counter counter-Schleife:
230 // d = Namens-Eintrag der Variable (p_name),
231 // son 0 = Startwert
232 // to: son 1 = Endwert, son 2 = NULL
233 // downto: son 1 = NULL, son 2 = Endwert
234 // son 3 = Schrittwert oder NULL
235 // son 4 = statements
236 // keyw_while while-Schleife: son 0 = cond, son 1 = statements
237 // keyw_repeat repeat-Schleife: son 0 = cond, son 1 = statements
238 //
239 // Vorrang-Klammern in Rechnungen und Bedingungen
240 // sind *keine* eigenen Knoten im Syntaxbaum
241 // sondern beeinflussen beim Parsen nur die Struktur des Syntaxbaumes
