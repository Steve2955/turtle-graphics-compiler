31type_t getTokenType(char *tok){
32 // check for operator-types
33 if (tok[0] == '^') return oper_pow;
34 if (tok[0] == '*') return oper_mul;
35 if (tok[0] == '/') return oper_div;
36 if (tok[0] == '+') return oper_add;
37 if (tok[0] == '|') return oper_abs;
38 if (tok[0] == '(') return oper_lpar;
39 if (tok[0] == ')') return oper_rpar;
40 if (tok[0] == ',') return oper_sep;
41 if (tok[0] == '=') return oper_equ;
42 if (tok[0] == '<') {
43 if (tok[1] == '>') return oper_nequ;
44 else if (tok[1] == '=') return oper_lequ;
45 else return oper_less;
46 }
47 if (tok[0] == '>') {
48 if (tok[1] == '=') return oper_gequ;
49 else return oper_grtr;
50 }
51
52 // check for const number
53 if (tok[0] == '.' || isdigit(tok[0])) {
54 bool hadDot = false; // a const number may contain at most one dot
55 int tokLen = strlen(tok);
56
57 for(int i = 0; i < tokLen; i++){
58 if(isdigit(tok[i]) || (tok[i] == '.' && !hadDot)){
59 hadDot = hadDot || tok[i] == '.';
60 }else{
61 fprintf(stderr, "Unzulässiger Zahlenwert in Zeile %d, Spalte %d\n", row, col);
62 exit(EXIT_FAILURE);
63 }
64 if(i == tokLen-1) return oper_const;
65 }
66 }
67
68 // Namenstabelle prüfen
69 for (int i = 0; i < nameCount; i++) {
70 if (strcmp(tok, name_tab[i].name) == 0) return name_tab[i].type;
71 }
72
73 //prüfen, ob zulässiger Variablen- oder Funktionsname, um es Namenstabelle hinzuzufügen
74 if (tok[0] == '_' || isalpha(tok[0]) || tok[0] == '@') {
75 for(int i = 0; tok[i] != '\0'; i++){
76 if(isalpha(tok[i]) || isdigit(tok[i]) || tok[i] == '_' || tok[0] == '@') {
77 continue;
78 }
79 else {
80 //kein zulässiger Variablen- oder Funktionsname -> Fehlermeldung & Abbruch
81 fprintf(stderr, "Unzulässiger Variablen- oder Funktionsname in Zeile %d, Spalte %d\n", row, col);
82 exit(EXIT_FAILURE);
83 }
84 }
85
86 if (nameCount > MAX_NAMES) {
87 fprintf(stderr, "Zu viele Variablen- und Funktionsnamen\n");
88 exit(EXIT_FAILURE);
89 }
90 type_t type = (tok[0] == '@') ? name_glob : name_any;
91 printf("type %d", type);
92 nameentry_t *name_entry = &(name_tab[nameCount]);
93 name_entry->type = type;
94
95 char *nameCopy = malloc(strlen(tok) + 1);
96 strcpy(nameCopy, tok);
97
98 name_entry->name = nameCopy;
99 nameCount++;
100 return type;
101 }
102
103 printf("Type not found: %s\n", tok);
104
105}

115void addToken(char *tok, type_t type){
116 // some logging
117 printf("Token: \"%s\" (%d:%d) \n", tok ? tok : "NONE", row, col);
118 // Create new token
119 token_t *tokPtr = firstTok->tok == NULL ? firstTok : malloc(sizeof(token_t));
120 // Create own copy of token string
121 if(tok != NULL){
122 char *tokCopy = malloc(strlen(tok) + 1);
123 strcpy(tokCopy, tok);
124 tokPtr->tok = tokCopy;
125 }
126 // save data to token
127 tokPtr->type = type;
128 tokPtr->pos.col = col;
129 tokPtr->pos.line = row;
130 // move pointers around
131 currentTok->next = tokPtr;
132 tokPtr->prev = currentTok;
133 currentTok = tokPtr;
134}

146token_t *readTokensFromFile(FILE *file){
147 row = col = 1;
148 // Buffer for current token
149 char *buf = malloc(MAX_TOKEN_LENGTH * sizeof(char));
150 // last read char
151 char c;
152 char lastC = '\n';
153 // length of current token
154 int len = 0;
155 // init token stream
156 initTokenStream();
157 // add bof to token stream
158 addToken(NULL, tok_bofeof);
159 // read char by char
160 while(lastC != EOF && (c = fgetc(file)) != EOF){
161 if(c == ' ' || c == '\t'){
162 // wird mit dem Whitespace ein Token abgeschlossen?
163 if(len > 0){
164 buf[len] = '\0';
165 type_t type = getTokenType((char *) buf);
166 addToken(buf, type);
167 // Spalte aktualisieren
168 col += len;
169 len = 0;
170 }
171 // Whitespaces werden ignoriert
172 col++;
173 continue;
174 }else if (isSpecial(c)){
175 // wird mit dem Sonderzeichen ein Token abgeschlossen?
176 if(len > 0){
177 buf[len] = '\0';
178 type_t type = getTokenType((char *) buf);
179 addToken(buf, type);
180 // Spalte aktualisieren
181 col += len;
182 len = 0;
183 }
184 // das Sonderzeichen selbst ist ein Token und muss behandelt werden
185
186 // kombinierte Sonderzeichen ("<=", ">=") werden zu einem Token zusammengefasst
187 // ToDo: gibt es noch mehr zusammengesetzte Operatoren?
188 if((lastC == '>' || lastC == '<') && c == '='){
189 revertToken(); //ToDo
190 // Buffer befüllen
191 buf[0] = lastC;
192 buf[1] = c;
193 buf[2] = '\0';
194 // <>=
195 type_t type = getTokenType((char *) buf);
196 addToken(buf, type);
197 // Spalte aktualisieren
198 col++;
199 }
200
201 // Sonderzeichen als Token hinzufügen
202 buf[0] = c;
203 buf[1] = '\0';
204 type_t type = getTokenType((char *) buf);
205 addToken(buf, type);
206
207 // Spalte aktualisieren
208 col++;
209 }else if (c == '\"' && lastC == '\n'){ // Kommentare werden ignoriert
210 while((c = fgetc(file)) != '\n' && c != EOF);
211 lastC = c;
212 row++; col=0;
213 }else if(c == '\n'){ // aktuelle Zeile ist beendet
214 // Falls nötig, schließe den Token vor dem newline ab
215 if(len > 0){
216 buf[len] = '\0';
217 type_t type = getTokenType((char *)buf);
218 addToken(buf, type);
219 }
220 // Zeile, Spalte und Länge zurücksetzen
221 row++; col = 0; len = 0;
222 }else{
223 // Spalte für nächsten Durchlauf aktualisieren
224 buf[len++] = c;
225 }
226 lastC = c;
227 }
228 // remaining token
229 if(len > 0){
230 buf[len] = '\0';
231 type_t type = getTokenType((char *) buf);
232 addToken(buf, type);
233 }
234
235 addToken(NULL, tok_bofeof);
236
237 // ToDo: einzelne Wörter auslesen -> Beispiel: "path circle(r,n)" wird zu "path|circle|(|r|,|n|)"
238 // ToDo: Wörter Typen zuordnen -> siehe getTokenType
239 // ToDo: Tokenliste verketten und zurückgeben
240 free(buf);
241 return firstTok;
242}
