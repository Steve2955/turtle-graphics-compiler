32  // check for operator-types
33  if (tok[0] == '^') return oper_pow;
34  if (tok[0] == '*') return oper_mul;
35  if (tok[0] == '/') return oper_div;
36  if (tok[0] == '+') return oper_add;
37  if (tok[0] == '-') return oper_sub;
38  if (tok[0] == '|') return oper_abs;
39  if (tok[0] == '(') return oper_lpar;
40  if (tok[0] == ')') return oper_rpar;
41  if (tok[0] == ',') return oper_sep;
42  if (tok[0] == '=') return oper_equ;
43  if (tok[0] == '<') {
44  if (tok[1] == '>') return oper_nequ;
45  else if (tok[1] == '=') return oper_lequ;
46  else return oper_less;
47  }
48  if (tok[0] == '>') {
49  if (tok[1] == '=') return oper_gequ;
50  else return oper_grtr;
51  }
52 
53  // check for const number
54  if (tok[0] == '.' || isdigit(tok[0])) {
55  bool hadDot = false; // a const number may contain at most one dot
56  int tokLen = strlen(tok);
57 
58  for(int i = 0; i < tokLen; i++){
59  if(isdigit(tok[i]) || (tok[i] == '.' && !hadDot)){
60  hadDot = hadDot || tok[i] == '.';
61  }else{
62  fprintf(stderr, "Unzulässiger Zahlenwert in Zeile %d, Spalte %d\n", row, col);
63  exit(EXIT_FAILURE);

65  if(i == tokLen-1) return oper_const;

67  }

69  // Namenstabelle prüfen
70  for (int i = 0; i < nameCount; i++) {
71  if (strcmp(tok, name_tab[i].name) == 0) return name_tab[i].type;
72  }
73 
74  //prüfen, ob zulässiger Variablen- oder Funktionsname, um es Namenstabelle hinzuzufügen
75  if (tok[0] == '_' || isalpha(tok[0]) || tok[0] == '@') {

148 token_t *readTokensFromFile(FILE *file){
149  row = col = 1;
150  // Buffer for current token
151  char *buf = malloc(MAX_TOKEN_LENGTH * sizeof(char));
152  // last read char
153  char c;
154  char lastC = '\n';
155  // length of current token
156  int len = 0;
157  // init token stream
158  initTokenStream();
159  // add bof to token stream
160  addToken(NULL, tok_bofeof);
161  // read char by char
162  while(lastC != EOF && (c = fgetc(file)) != EOF){
163  if(c == ' ' || c == '\t'){
164  // wird mit dem Whitespace ein Token abgeschlossen?
165  if(len > 0){
166  buf[len] = '\0';
167  type_t type = getTokenType((char *) buf);
168  addToken(buf, type);
169  // Spalte aktualisieren
170  col += len;
171  len = 0;
172  }
173  // Whitespaces werden ignoriert
174  col++;
175  continue;
176  }else if (isSpecial(c)){
177  // wird mit dem Sonderzeichen ein Token abgeschlossen?
178  if(len > 0){
179  buf[len] = '\0';
180  type_t type = getTokenType((char *) buf);
181  addToken(buf, type);
182  // Spalte aktualisieren
183  col += len;
184  len = 0;
185  }
186  // das Sonderzeichen selbst ist ein Token und muss behandelt werden
187 
188  // kombinierte Sonderzeichen ("<=", ">=") werden zu einem Token zusammengefasst
189  // ToDo: gibt es noch mehr zusammengesetzte Operatoren?
190  if((lastC == '>' || lastC == '<') && c == '='){
191  revertToken(); //ToDo
192  // Buffer befüllen
193  buf[0] = lastC;
194  buf[1] = c;
195  buf[2] = '\0';
196  // <>=
197  type_t type = getTokenType((char *) buf);
198  addToken(buf, type);
199  // Spalte aktualisieren
200  col++;
201  }
202 
203  // Sonderzeichen als Token hinzufügen
204  buf[0] = c;
205  buf[1] = '\0';
206  type_t type = getTokenType((char *) buf);
207  addToken(buf, type);
208 
209  // Spalte aktualisieren
210  col++;
211  }else if (c == '\"' && lastC == '\n'){ // Kommentare werden ignoriert
212  while((c = fgetc(file)) != '\n' && c != EOF);
213  lastC = c;
214  row++; col=0;
215  }else if(c == '\n'){ // aktuelle Zeile ist beendet
216  // Falls nötig, schließe den Token vor dem newline ab
217  if(len > 0){
218  buf[len] = '\0';
219  type_t type = getTokenType((char *)buf);
220  addToken(buf, type);
221  }
222  // Zeile, Spalte und Länge zurücksetzen
223  row++; col = 0; len = 0;
224  }else{
225  // Spalte für nächsten Durchlauf aktualisieren
226  buf[len++] = c;
227  }
228  lastC = c;
229  }
230  // remaining token
231  if(len > 0){
232  buf[len] = '\0';
233  type_t type = getTokenType((char *) buf);
234  addToken(buf, type);
235  }
236 
237  addToken(NULL, tok_bofeof);
238

243  return firstTok;
